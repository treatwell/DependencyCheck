108     /**
109      * Performs validation on the configuration to ensure that the correct
110      * analyzers are in place.
111      *
112      * @param engine the dependency-check engine
113      * @throws InitializationException thrown if there is a configuration error
114      */
115     @Override
116     protected void prepareFileTypeAnalyzer(Engine engine) throws InitializationException {
117         if (engine.getMode() != Mode.EVIDENCE_COLLECTION) {
118             try {
119                 final Settings settings = engine.getSettings();
120                 final String[] tmp = settings.getArray(Settings.KEYS.ECOSYSTEM_SKIP_CPEANALYZER);
121                 if (tmp != null) {
122                     final List<String> skipEcosystems = Arrays.asList(tmp);
123                     if (skipEcosystems.contains(DEPENDENCY_ECOSYSTEM)
124                             && !settings.getBoolean(Settings.KEYS.ANALYZER_OSSINDEX_ENABLED)) {
125                         if (!settings.getBoolean(Settings.KEYS.ANALYZER_NODE_AUDIT_ENABLED)) {
126                             final String msg = "Invalid Configuration: enabling the Node Package Analyzer without "
127                                     + "using the Node Audit Analyzer or OSS Index Analyzer is not supported.";
128                             throw new InitializationException(msg);
129                         } else if (!isNodeAuditEnabled(engine)) {
130                             final String msg = "Missing package.lock or npm-shrinkwrap.lock file: Unable to scan a node "
131                                     + "project without a package-lock.json or npm-shrinkwrap.json.";
132                             throw new InitializationException(msg);
133                         }
134                     } else if (skipEcosystems.contains(DEPENDENCY_ECOSYSTEM)
135                             && !settings.getBoolean(Settings.KEYS.ANALYZER_NODE_AUDIT_ENABLED)) {
136                         LOGGER.warn("Using only the OSS Index Analyzer with Node.js can result in many false positives "
137                                 + "- please enable the Node Audit Analyzer.");
138                     }
139                 }
140             } catch (InvalidSettingException ex) {
141                 throw new InitializationException("Unable to read configuration settings", ex);
142             }
143         }
144     }

177     /**
178      * Determines if the Node Audit analyzer is enabled.
179      *
180      * @param engine a reference to the dependency-check engine
181      * @return <code>true</code> if the Node Audit Analyzer is enabled;
182      * otherwise <code>false</code>
183      */
184     private boolean isNodeAuditEnabled(Engine engine) {
185         for (Analyzer a : engine.getAnalyzers()) {
186             if (a instanceof NodeAuditAnalyzer) {
187                 if (a.isEnabled()) {
188                     try {
189                         ((NodeAuditAnalyzer) a).prepareFileTypeAnalyzer(engine);
190                     } catch (InitializationException ex) {
191                         LOGGER.debug("Error initializing the NodeAuditAnalyzer");
192                     }
193                 }
194                 return a.isEnabled();
195             }
196         }
197         return false;
198     }

200     /**
201      * Checks if a package lock file or equivalent exists for the NPM project.
202      *
203      * @param dependencyFile a reference to the `package.json` file
204      * @return <code>true</code> if no lock file is found; otherwise
205      * <code>true</code>
206      */
207     private boolean noLockFileExists(File dependencyFile) {
208         //TODO if we support yarn we need to add the check for the yarn.yml file
209         final File lock = new File(dependencyFile.getParentFile(), "package-lock.json");
210         final File shrinkwrap = new File(dependencyFile.getParentFile(), "npm-shrinkwrap.json");
211         return !(lock.isFile() || shrinkwrap.isFile());
212     }

214     @Override
215     protected void analyzeDependency(Dependency dependency, Engine engine) throws AnalysisException {
216         if (isNodeAuditEnabled(engine)
217                 && !(PACKAGE_LOCK_JSON.equals(dependency.getFileName()) || SHRINKWRAP_JSON.equals(dependency.getFileName()))) {
218             engine.removeDependency(dependency);
219         }
220         final File dependencyFile = dependency.getActualFile();
221         if (!dependencyFile.isFile() || dependencyFile.length() == 0 || !shouldProcess(dependencyFile)) {
222             return;
223         }
224         if (noLockFileExists(dependency.getActualFile())) {
225             LOGGER.warn("No lock file exists - this will result in false negatives; please run `npm install --package-lock`");
226         }
227         final File baseDir = dependencyFile.getParentFile();
228         if (PACKAGE_JSON.equals(dependency.getFileName())) {
229             final File lockfile = new File(baseDir, PACKAGE_LOCK_JSON);
230             final File shrinkwrap = new File(baseDir, SHRINKWRAP_JSON);
231             if (shrinkwrap.exists() || lockfile.exists()) {
232                 return;
233             }
234         } else if (PACKAGE_LOCK_JSON.equals(dependency.getFileName())) {
235             final File shrinkwrap = new File(baseDir, SHRINKWRAP_JSON);
236             if (shrinkwrap.exists()) {
237                 return;
238             }
239         }
240         final File nodeModules = new File(baseDir, "node_modules");
241         if (!nodeModules.isDirectory()) {
242             LOGGER.warn("Analyzing `{}` - however, the node_modules directory does not exist. "
243                     + "Please run `npm install` prior to running dependency-check", dependencyFile.toString());
244             return;
245         }
246 
247         try (JsonReader jsonReader = Json.createReader(FileUtils.openInputStream(dependencyFile))) {
248             final JsonObject json = jsonReader.readObject();
249             final String parentName = json.getString("name", "");
250             final String parentVersion = json.getString("version", "");
251             if (parentName.isEmpty()) {
252                 return;
253             }
254             dependency.setName(parentName);
255             final String parentPackage;
256             if (!parentVersion.isEmpty()) {
257                 dependency.setVersion(parentVersion);
258                 parentPackage = String.format("%s:%s", parentName, parentVersion);
259             } else {
260                 parentPackage = parentName;
261             }
262             processDependencies(json, baseDir, dependencyFile, parentPackage, engine);
263         } catch (JsonException e) {
264             LOGGER.warn("Failed to parse package.json file.", e);
265         } catch (IOException e) {
266             throw new AnalysisException("Problem occurred while reading dependency file.", e);
267         }
268     }

270     /**
271      * Process the dependencies in the lock file by first parsing its
272      * dependencies and then finding the package.json for the module and adding
273      * it as a dependency.
274      *
275      * @param json the data to process
276      * @param baseDir the base directory being scanned
277      * @param rootFile the root package-lock/npm-shrinkwrap being analyzed
278      * @param parentPackage the parent package name of the current node
279      * @param engine a reference to the dependency-check engine
280      * @throws AnalysisException thrown if there is an exception
281      */
282     private void processDependencies(JsonObject json, File baseDir, File rootFile,
283             String parentPackage, Engine engine) throws AnalysisException {
284         if (json.containsKey("dependencies")) {
285             final JsonObject deps = json.getJsonObject("dependencies");
286             for (Map.Entry<String, JsonValue> entry : deps.entrySet()) {
287                 final JsonObject jo = (JsonObject) entry.getValue();
288                 final String name = entry.getKey();
289                 final String version = jo.getString("version");
290                 final boolean optional = jo.getBoolean("optional", false);
291                 final File base = Paths.get(baseDir.getPath(), "node_modules", name).toFile();
292                 final File f = new File(base, PACKAGE_JSON);
293 
294                 if (optional && !f.exists()) {
295                     LOGGER.warn("node module {} seems optional and not installed, skip it", name);
296                     continue;
297                 }
298 
299                 if (jo.containsKey("dependencies")) {
300                     final String subPackageName = String.format("%s/%s:%s", parentPackage, name, version);
301                     processDependencies(jo, base, rootFile, subPackageName, engine);
302                 }
303 
304                 final Dependency child;
305                 if (f.exists()) {
306                     //TOOD - we should use the integrity value instead of calculating the SHA1/MD5
307                     child = new Dependency(f);
308                     child.setEcosystem(DEPENDENCY_ECOSYSTEM);
309 
310                     try (JsonReader jr = Json.createReader(FileUtils.openInputStream(f))) {
311                         final JsonObject childJson = jr.readObject();
312                         gatherEvidence(childJson, child);
313 
314                     } catch (JsonException e) {
315                         LOGGER.warn("Failed to parse package.json file from dependency.", e);
316                     } catch (IOException e) {
317                         throw new AnalysisException("Problem occurred while reading dependency file.", e);
318                     }
319                 } else {
320                     LOGGER.warn("Unable to find node module: {}", f.toString());
321                     child = new Dependency(rootFile, true);
322                     child.setEcosystem(DEPENDENCY_ECOSYSTEM);
323                     //TOOD - we should use the integrity value instead of calculating the SHA1/MD5
324                     child.setSha1sum(Checksum.getSHA1Checksum(String.format("%s:%s", name, version)));
325                     child.setSha256sum(Checksum.getSHA256Checksum(String.format("%s:%s", name, version)));
326                     child.setMd5sum(Checksum.getMD5Checksum(String.format("%s:%s", name, version)));
327                     child.addEvidence(EvidenceType.VENDOR, rootFile.getName(), "name", name, Confidence.HIGHEST);
328                     child.addEvidence(EvidenceType.PRODUCT, rootFile.getName(), "name", name, Confidence.HIGHEST);
329                     child.addEvidence(EvidenceType.VERSION, rootFile.getName(), "version", version, Confidence.HIGHEST);
330                     child.setName(name);
331                     child.setVersion(version);
332                     final String packagePath = String.format("%s:%s", name, version);
333                     child.setDisplayFileName(packagePath);
334                     child.setPackagePath(packagePath);
335                     try {
336                         final PackageURL purl = PackageURLBuilder.aPackageURL().withType("npm").withName(name).withVersion(version).build();
337                         final PurlIdentifier id = new PurlIdentifier(purl, Confidence.HIGHEST);
338                         child.addSoftwareIdentifier(id);
339                     } catch (MalformedPackageURLException ex) {
340                         LOGGER.debug("Unable to build package url for `" + packagePath + "`", ex);
341                     }
342                 }
343 
344                 child.addProjectReference(parentPackage);
345                 child.setEcosystem(DEPENDENCY_ECOSYSTEM);
346 
347                 final Dependency existing = findDependency(engine, name, version);
348                 if (existing != null) {
349                     if (existing.isVirtual()) {
350                         DependencyMergingAnalyzer.mergeDependencies(child, existing, null);
351                         engine.removeDependency(existing);
352                         engine.addDependency(child);
353                     } else {
354                         DependencyBundlingAnalyzer.mergeDependencies(existing, child, null);
355                     }
356                 } else {
357                     engine.addDependency(child);
358                 }
359             }
360         }
361     }
